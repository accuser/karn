  case "observes the seam" {
    let r <- authorise.call(100)
    expect Logger.log called

    expect Logger.log called 2 times
    expect Logger.log called with msg == "hi"
    expect Logger.log called 1 times with amount > 0
    expect Logger.log never called
    expect Logger.log before Audit.record
    let calls = trace(Logger.log)
    expect calls.length() == 1
  }
}
